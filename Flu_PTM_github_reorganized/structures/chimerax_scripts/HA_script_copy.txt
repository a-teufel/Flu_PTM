# Positive selection sites (red)

color sel red
show sel atoms
style sel sphere
size sel atomRadius 1.5

label sel offset 4.0,4.0,2.0
label sel bgColor white
label sel outline true
label sel outlineWidth 2

# Add leader lines pointing to the residues
2dlabels create line color red width 3 start atom #1:183@CA end label #1:183
2dlabels create line color red width 3 start atom #1:571@CA end label #1:571
